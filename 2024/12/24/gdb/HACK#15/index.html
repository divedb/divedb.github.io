测试程序

#include <linux/init.h>
#include <linux/module.h>

static int init_oopsdemo() {
  printk(KERN_ALERT "Inside %s", __FUNCTION__);
  *((int*)0x00) = 0x19760817;
  return 0;
}

  printk(KERN_ALERT "Inside %s", __FUNCTION__);
